function nscaler(a) {
    var b = "";
    var ar = String(a).split('');
    $.each(ar, function (i, e) {
        switch (e) {
        case "0":
            b += "0";
            break;
        case "1":
            b += "2";
            break;
        case "2":
            b += "5";
            break;
        case "3":
